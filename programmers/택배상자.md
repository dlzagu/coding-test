```js
function solution(order) {
  const Queue = new Array(order.length).fill().map((_, i) => i + 1);
  const stack = [];
  let cnt = 0;

  while (Queue.length) {
    if (order[cnt] !== Queue[0]) {
      if (stack.length && order[cnt] === stack.at(-1)) {
        cnt++;
        stack.pop();
      } else {
        stack.push(Queue.shift());
      }
    } else {
      cnt++;
      Queue.shift();
    }
  }

  while (stack.length) {
    if (order[cnt] == stack.at(-1)) {
      cnt++;
      stack.pop();
    } else break;
  }

  return cnt;
}
```
